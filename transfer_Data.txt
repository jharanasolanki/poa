;transfer data from one segment to another

DATA SEGMENT
SEG1 DB 1H, 2H, 3H
ENDS 

EXTRA SEGMENT 
SEG2 DB ?
ENDS

CODE SEGMENT
START :
; SET SEGMENT REGISTERS
MOV AX, DATA
MOV DS, AX
MOV AX, EXTRA
MOV ES, AX

LEA SI , SEG1
LEA DI , SEG2
MOV CX , 03H
X: MOV AH, DS: [SI]
MOV ES:[DI], AH
INC SI
INC DI
DEC CX
JNZ X INT 3
ENDS
END START
